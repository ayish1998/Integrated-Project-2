Data Visualization: Practical How-Tos for Presentation and Analysis

This tutorial shows how to use Trending Job Markets API and visualize with D3.js and Observable to create interactive and dynamic data visualizations such as charts and maps

Overview

Introduction

Welcome to the tutorial on how to use APIs and D3.js to visualize data! In this step-by-step tutorial, we will explore how to fetch data from APIs, process and manipulate the data, and then use D3.js to create interactive and dynamic data visualizations such as charts and maps. By the end of this tutorial, you will be able to create compelling visualizations to gain insights and present data in a more engaging manner.

Job Data Page Overview

The job data page offers a comprehensive display of real-time job market insights, presenting users with current employment statistics, historical trends, sector analysis, and interactive visualizations. API Access

Step 1: Fetch Data via API URL

To access data from a live API, follow these steps:

  • Sign up for an API key, which you'll need to make requests and retrieve data.
  • The API offers various endpoints, including:
    • REST COUNTRIES - RESTful API: Get detailed country information via a RESTful API. Current version: 3.1. Inspired by restcountries.eu, it provides comprehensive country data.
    • REST COUNTRIES 🇵🇪 - Country Information: Fetch detailed country data through this RESTful API, offering up-to-date insights about nations globally.
    • RAPID API - Job Search Data Visualization: Retrieve job search data for visualization, including job details, estimated salaries, and search filters.
    • GET Search - Search Filters for Job Data: Refine job-related information with search filters, enhancing user search experience.
    • Search Filters - Detailed Job Search Filters: Access comprehensive filters for precise job searches based on criteria and preferences.
    • Job Details - In-Depth Job Information: Obtain comprehensive job details to make informed decisions during job searches.
    • Estimated Salary - Projected Salary Information: Retrieve salary estimates for different job listings, providing users with insights into potential earnings.

async function getCountryList() {
    const url = 'https://restcountries.com/v3.1/all';
    try {
        const response = await fetch(url);
        const data = await response.json();
        return data.map((country) => country.name.common);
    } catch (error) {
        console.error(error);
    }
}

// Function to retrieve job data from the API
async function getJobData(selectedCountry) {
    const countryList = await getCountryList();
    const country = countryList.find(
        (country) => country === toTitleCase(selectedCountry)
    );

    if (!country) {
        console.error('Country not found.');
        return null;
    }

    const formattedCountry = encodeURIComponent(country);
    const jobTitle = 'NodeJS Developer';

    const jobUrl = `https://jsearch.p.rapidapi.com/search?query=${formattedCountry}&page=1&num_pages=1`;
    const salaryUrl = `https://jsearch.p.rapidapi.com/estimated-salary?job_title=${encodeURIComponent(
        jobTitle
    )}&location=${formattedCountry}&radius=100`;
    const jobOptions = {
        method: 'GET',
        headers: {
            'X-RapidAPI-Key': 'YOUR_RAPIDAPI_KEY',
            'X-RapidAPI-Host': 'jsearch.p.rapidapi.com'
        },
    };

    const [jobResponse, salaryResponse] = await Promise.all([
        fetch(jobUrl, jobOptions),
        fetch(salaryUrl, jobOptions),
    ]);

    const jobData = await jobResponse.json();
    const salaryData = await salaryResponse.json();

    const combinedData = jobData.data.map((job) => ({
        ...job,
        estimated_salary: findEstimatedSalary(job.job_title, salaryData.data),
    }));

    return combinedData;
}

Replace 'YOUR_RAPIDAPI_KEY' with your actual RapidAPI key. This code snippet retrieves job and salary data based on the selected country using the Rest Countries and Rapid API, processes and combines the data, and returns it for rendering.

function renderChart(data) {
    const chartContainer = document.getElementById('chart-container');
    chartContainer.innerHTML = ''; // Clear previous chart

    if (!data || data.length === 0) {
        chartContainer.innerHTML = '
No data available for chart.
';
        return;
    }

    const margin = { top: 20, right: 30, bottom: 40, left: 40 };
    const width = 600 - margin.left - margin.right;
    const height = 400 - margin.top - margin.bottom;

    const svg = d3.select(chartContainer)
        .append('svg')
        .attr('width', width + margin.left + margin.right)
        .attr('height', height + margin.top + margin.bottom)
        .append('g')
        .attr('transform', `translate(${margin.left},${margin.top})`);

    const x = d3.scaleBand()
        .domain(data.map(job => job.job_title))
        .range([0, width])
        .padding(0.1);

    const y = d3.scaleLinear()
        .domain([0, d3.max(data, job => job.estimated_salary)])
        .nice()
        .range([height, 0]);

    svg.selectAll('.bar')
        .data(data)
        .enter().append('rect')
        .attr('class', 'bar')
        .attr('x', job => x(job.job_title))
        .attr('width', x.bandwidth())
        .attr('y', job => y(job.estimated_salary))
        .attr('height', job => height - y(job.estimated_salary));

    svg.append('g')
        .attr('class', 'x-axis')
        .attr('transform', `translate(0,${height})`)
        .call(d3.axisBottom(x));

    svg.append('g')
        .attr('class', 'y-axis')
        .call(d3.axisLeft(y).ticks(5));

              // Function to filter job data based on keyword
              function filterJobsByKeyword(keyword, data) {
                  const filteredData = data.filter(job => {
                      return (
                          job.job_title.toLowerCase().includes(keyword.toLowerCase()) ||
                          job.job_description.toLowerCase().includes(keyword.toLowerCase()) ||
                          job.job_required_skills.toLowerCase().includes(keyword.toLowerCase())
                      );
                  });
                  return filteredData;
              }
              
              // Event listener for the search button
              const searchButton = document.getElementById('search-button');
              searchButton.addEventListener('click', () => {
                  const searchInput = document.getElementById('search-input');
                  const keyword = searchInput.value;
                  const filteredData = filterJobsByKeyword(keyword, jobData);
                  renderJobData(filteredData);
                  renderChart(filteredData);
              });
              // Function to render paginated job data
              function renderPaginatedJobData(data, page, itemsPerPage) {
                  const startIndex = (page - 1) * itemsPerPage;
                  const endIndex = startIndex + itemsPerPage;
                  const paginatedData = data.slice(startIndex, endIndex);
                  renderJobData(paginatedData);
              }
              
              // Event listener for page navigation
              const prevPageButton = document.getElementById('prev-page-button');
              const nextPageButton = document.getElementById('next-page-button');
              let currentPage = 1;
              const itemsPerPage = 10;
              
              prevPageButton.addEventListener('click', () => {
                  if (currentPage > 1) {
                      currentPage--;
                      renderPaginatedJobData(jobData, currentPage, itemsPerPage);
                  }
              });
              
              nextPageButton.addEventListener('click', () => {
                  const totalPages = Math.ceil(jobData.length / itemsPerPage);
                  if (currentPage < totalPages) {
                      currentPage++;
                      renderPaginatedJobData(jobData, currentPage, itemsPerPage);
                  }
              });
              // Event listener for sorting options
              const sortSelect = document.getElementById('sort-select');
              sortSelect.addEventListener('change', () => {
                  const sortBy = sortSelect.value;
                  let sortedData;
              
                  if (sortBy === 'title') {
                      sortedData = jobData.slice().sort((a, b) => a.job_title.localeCompare(b.job_title));
                  } else if (sortBy === 'salary') {
                      sortedData = jobData.slice().sort((a, b) => b.estimated_salary - a.estimated_salary);
                  } else if (sortBy === 'location') {
                      sortedData = jobData.slice().sort((a, b) => a.job_country.localeCompare(b.job_country));
                  }
              
                  renderPaginatedJobData(sortedData, currentPage, itemsPerPage);
              });

        .fitSize([width, height], geojson);

      // Create a path generator
      const path = d3.geoPath().projection(projection);

      // Create color scale
      const colorScale = d3.scaleSequential(d3.interpolateBlues)
        .domain([0, d3.max(data, d => d.value)]);

      // Draw map features
      svg.selectAll("path")
        .data(geojson.features)
        .enter().append("path")
        .attr("d", path)
        .attr("fill", d => colorScale(data.find(item => item.name === d.properties.name).value));
    });
